 function rgb(r, g, b) {
  const makeHex = (num) => {
   if (num < 0 ) {
    return "00"; 
   } else if (num > 255) {
    return "FF";
   }

   return ('0' + Number(num).toString(16)).slice(-2).toUpperCase();
  }

  return makeHex(r) + makeHex(g) + makeHex(b);
 }
